import { extractTextFromImage } from "../utils/ocr";

export const analyzeDocuments = async (files, modes) => {

  // 👉 PROCESSA OS ARQUIVOS COM OCR
  const processedFiles = await Promise.all(
    files.map(async (file) => {
      let text = "";

      // 👉 se não tem texto → OCR
      if (!text || text.trim().length < 50) {
        console.log("Aplicando OCR:", file.name);
        console.log("Texto extraído:", text.substring(0, 200));

        try {
          import { extractTextFromPDF } from "../utils/ocr";

text = await extractTextFromPDF(file.raw || file);
        } catch (e) {
          console.error("Erro OCR:", e);
        }
      }

      return {
        name: file.name,
        text
      };
    })
  );

  // 👉 CHAMA SUA API NORMAL
  const response = await fetch("/api/analyze", {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ files: processedFiles, modes })
  });

  if (!response.ok) {
    const err = await response.json();
    throw new Error(err.error || "Erro na análise");
  }

  return await response.json();
};
